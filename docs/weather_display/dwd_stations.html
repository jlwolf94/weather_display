102    def process_stations_page(response):
103        """
104        Method that processes the response content of a request to the standard url
105        and returns a formatted list of the stations with their informations.
106
107        Parameters
108        ----------
109        response (Response):
110            A response object retrieved from a request to the standard url.
111
112        Returns
113        -------
114        table_entries (list):
115            A list containing all rows from the stations table as
116            individual dictionaries.
117        """
118
119        # Convert the response content to a searchable object.
120        stations_page = BeautifulSoup(response.content, features="lxml")
121
122        # Extract the table from the page and convert it to a json file.
123        table = stations_page.find("table")
124        column_names = []
125        table_entries = []
126
127        # Find all column names.
128        for tr in table.find_all("tr", recursive=False):
129            for th in tr.find_all("th", recursive=False):
130                if th.text != "Stationslexikon":
131                    column_names.append(th.text)
132
133        # Find all column entries and add them to the list of entries.
134        for tr in table.find_all("tr", recursive=False):
135            entry = {}
136            for i, td in enumerate(tr.find_all("td", recursive=False)):
137                entry[column_names[i]] = str(td.text).replace(u"\xa0", u" ")
138            if entry:
139                table_entries.append(entry)
140
141        return table_entries

226        """
227        Method that searches the station closest to the current position which
228        is represented by the latitude and longitude value. The informations
229        of the closest station with hourly measurements are returned.
230
231        Parameters
232        ----------
233        latitude (float):
234            The current geographic latitude.
235
236        longitude (float):
237            The current geographic longitude.
238
239        Returns
240        -------
241        station_info (dict):
242            A dictionary containing all informations of the station or
243            an empty dictionary if no station is found.
244        """
245
246        # Check whether data for a search is available.
247        if self.table_entries:
248            # Get stations with automated hourly measurements.
249            stations = [st for st in self.table_entries if st["Kennung"] == "SY"]
250
251            # Search for the closest station.
252            min_distance = float("inf")
253            min_station = {}
254            for st in stations:
255                curr_distance = math.dist((latitude, longitude),
256                                          (float(st["Breite"]), float(st["Länge"])))
257                if curr_distance < min_distance:
258                    min_distance = curr_distance
259                    min_station = st
260
261            return min_station
262        else:
263            return {}
