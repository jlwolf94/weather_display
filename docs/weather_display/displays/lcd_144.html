weather_display.displays.lcd_144

The lcd_144 module contains the LCD144 class that is responsible for all input, output and setting methods for the 1.44inch LCD HAT SPI interface from Waveshare.

Functions with the ST7735S chip driver to clear the screen, write, draw lines, draw in general and to perform other functions.

260    def set_gram_scan_way(self, scan_dir):
261        """
262        Method that sets the display scan direction.
263
264        Parameters
265        ----------
266        scan_dir (int):
267            Number for the new scan direction.
268        """
269
270        # Save the new screen scan direction
271        self.scan_dir = scan_dir
272
273        # Get GRAM and LCD width and height
274        if ((scan_dir == L2R_U2D) or (scan_dir == L2R_D2U)
275            or (scan_dir == R2L_U2D) or (scan_dir == R2L_D2U)):
276            self.width = LCD_HEIGHT
277            self.height = LCD_WIDTH
278            if scan_dir == L2R_U2D:
279                memory_access_reg_data = 0X00 | 0x00
280            elif scan_dir == L2R_D2U:
281                memory_access_reg_data = 0X00 | 0x80
282            elif scan_dir == R2L_U2D:
283                memory_access_reg_data = 0x40 | 0x00
284            else:
285                # R2L_D2U:
286                memory_access_reg_data = 0x40 | 0x80
287        else:
288            self.width = LCD_WIDTH
289            self.height = LCD_HEIGHT
290            if scan_dir == U2D_L2R:
291                memory_access_reg_data = 0X00 | 0x00 | 0x20
292            elif scan_dir == U2D_R2L:
293                memory_access_reg_data = 0X00 | 0x40 | 0x20
294            elif scan_dir == D2U_L2R:
295                memory_access_reg_data = 0x80 | 0x00 | 0x20
296            else:
297                # R2L_D2U:
298                memory_access_reg_data = 0x40 | 0x80 | 0x20
299
300        # Please set (memory_access_reg_data & 0x10) != 1
301        if (memory_access_reg_data & 0x10) != 1:
302            self.x_adjust = LCD_Y
303            self.y_adjust = LCD_X
304        else:
305            self.x_adjust = LCD_X
306            self.y_adjust = LCD_Y
307
308        # Set the read and write scan direction of the frame memory
309        # MX, MY, RGB mode
310        self.write_reg(0x36)
311        if LCD_1IN44 == 1:
312            # 0x08 set RGB
313            self.write_data_8bit(memory_access_reg_data | 0x08)
314        else:
315            # RGB color filter panel
316            self.write_data_8bit(memory_access_reg_data & 0xf7)

318    def init_LCD(self, scan_dir):
319        """
320        Method that initializes the whole display.
321
322        Parameters
323        ----------
324        scan_dir (int):
325            Number for the new scan direction.
326
327        Returns
328        -------
329        success (Optional[int]):
330            On failure -1 is returned and None in all other cases.
331        """
332
333        if (self.config.init_GPIO() != 0):
334            return -1
335
336        # Turn on the backlight
337        GPIO.output(self.config.LCD_BL_PIN, GPIO.HIGH)
338
339        # Hardware reset
340        self.reset()
341
342        # Set the initialization register
343        self.init_reg()
344
345        # Set the display scan and color transfer modes
346        self.set_gram_scan_way(scan_dir)
347        self.config.delay_driver_ms(200)
348
349        # Sleep out
350        self.write_reg(0x11)
351        self.config.delay_driver_ms(120)
352
353        # Turn on the LCD display
354        self.write_reg(0x29)

356    def set_windows(self, x_start, y_start, x_end, y_end):
357        """
358        Method that sets the start position and size of the display area.
359
360        Parameters
361        ----------
362        x_start (int):
363            Start position in x-direction.
364
365        y_start (int):
366            Start position in y-direction.
367
368        x_end (int):
369            End position in x-direction.
370
371        y_end (int):
372            End position in y-direction.
373        """
374
375        # Set the x-coordinates
376        self.write_reg(0x2A)
377        self.write_data_8bit(0x00)
378        self.write_data_8bit((x_start & 0xff) + self.x_adjust)
379        self.write_data_8bit(0x00)
380        self.write_data_8bit(((x_end - 1) & 0xff) + self.x_adjust)
381
382        # Set the y-coordinates
383        self.write_reg(0x2B)
384        self.write_data_8bit(0x00)
385        self.write_data_8bit((y_start & 0xff) + self.y_adjust)
386        self.write_data_8bit(0x00)
387        self.write_data_8bit(((y_end - 1) & 0xff) + self.y_adjust)
388
389        self.write_reg(0x2C)

391    def clear(self):
392        """
393        Method that clears the display.
394        """
395
396        buffer = [0xff] * (self.width * self.height * 2)
397        self.set_windows(0, 0, self.width, self.height)
398        GPIO.output(self.config.LCD_DC_PIN, GPIO.HIGH)
399        for i in range(0, len(buffer), 4096):
400            self.config.write_SPI_bytes(buffer[i:i+4096])

402    def show_image(self, image):
403        """
404        Method that shows an image on the display with the already configurated
405        settings.
406
407        Parameters
408        ----------
409        image (Any):
410            Image that will be shown on the display.
411            The image needs to match the display size in pixel.
412        """
413
414        # Check whether an image exists.
415        if (image == None):
416            return
417
418        # Get the image dimensions and compare them with the display size.
419        img_width, img_height = image.size
420        if img_width != self.width or img_height != self.height:
421            raise ValueError(
422                f"Image must be same dimensions as the display ({self.width}x{self.height}).")
423
424        img = np.asarray(image)
425        pix = np.zeros((self.width, self.height, 2), dtype=np.uint8)
426        pix[..., [0]] = np.add(np.bitwise_and(img[..., [0]], 0xF8),
427                               np.right_shift(img[..., [1]], 5))
428        pix[..., [1]] = np.add(np.bitwise_and(np.left_shift(img[..., [1]], 3), 0xE0),
429                               np.right_shift(img[..., [2]], 3))
430        pix = pix.flatten().tolist()
431        self.set_windows(0, 0, self.width, self.height)
432        GPIO.output(self.config.LCD_DC_PIN, GPIO.HIGH)
433        for i in range(0, len(pix), 4096):
434            self.config.write_SPI_bytes(pix[i:i+4096])
